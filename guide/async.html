# Async loading

In some case you might have a very large amount of nodes to display. In this case you can to use "async loading" and update treeview with data coming from the server and have infinite nodes.

WARNING

When using async loading, checkbox mode auto will reload all parents state. If the node coming from the server is unchecked it will uncheck all parents when added

<template>
  <Tree :nodes="nodes" :config="config" @node-opened="addServerNode">
    <template #loading-slot>
      <div class="progress">
        <div class="indeterminate"></div>
      </div>
    </template>
  </Tree>
</template>

<script lang="ts">
import Tree from "@src/components/Tree.vue";
import { INode } from "/@src/structure/INode";

export default {
  components: {
    Tree,
  },
  data: function () {
    return {
      nodes: {
        id1: {
          text: "text1",
          children: ["id11", "id12"],
        },
        id11: {
          text: "text11",
        },
        id12: {
          text: "text12",
        },
        id2: {
          text: "text2",
        },
      },
      config: {
        roots: ["id1", "id2"],
        leaves: ["fakeid"],
      },
    };
  },
  methods: {
    addServerNode(n: INode): void {
      if (n.children && n.children.length > 0) return;

      // set node loading state to tree
      n.state.isLoading = true;

      // fake server call
      setTimeout(() => {
        // create a fake node
        const id = `${Date.now()}`;
        const newNode = {
          text: `loaded from server`,
          children: [],
          state: {},
        };

        // add the node to nodes
        this.nodes[id] = newNode;
        // set children
        n.children = [id];
        // end loading
        n.state.isLoading = false;
      }, 2000);
    },
  },
};
</script>
